Feedback

SELECT R1.dateR, nomP, nomV
FROM LesRepas R1 JOIN LesRepas R2 ON (R1.dateR = R2.dateR)
                 JOIN LeMenu ON (R1.dateR = LeMenu.dateR)
WHERE R1.nomI = 'Marie' AND R2.nomI = 'Adrian';


--Variant intersect
SELECT dateR, nomP, nomV 
FROM LesRepas JOIN LeMenu USING(dateR) 
WHERE nomI = 'Marie'

FROM LesRepas JOIN LeMenu USING(dateR) 
WHERE nomI ='Adrian';
